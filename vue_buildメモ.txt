# vue cli3 をグローバルにインストール
npm install -g @vue/cli

# vue project作成 (ファイル共有していない場所で)
vue create my-project
# node_modulesファイル削除
rm node_modules
# プロジェクトへ移動
cd my-project dist

# node_modulesの場所変更(vagrantの場合)
mkdir -p ~slim_app_node_modules/node_modules
ln -s ~slim_app_node_modules/node_modules node_modules

# 設定ファイルは vue.config.js

# ビルド
npm run build
